CREATE TABLE Login(User_id INT(10) PRIMARY KEY AUTO_INCREMENT, Username VARCHAR(15) UNIQUE, User_password VARCHAR(10), NAME VARCHAR(20), User_role VARCHAR(10));

CREATE TABLE Customer(Customer_Num INT PRIMARY KEY,Name VARCHAR(50) NOT NULL,Gender CHAR(1) NOT NULL,DOB DATE NOT NULL,Address VARCHAR(70) NOT NULL,Pincode INT(10) NOT NULL,Contact_Number BIGINT(19) NOT NULL,Mother_Name VARCHAR(150) NOT NULL,Father_Name VARCHAR(150) NOT NULL,Marital_status CHAR(10) NOT NULL,FOREIGN KEY(Customer_Num) REFERENCES login(User_Id));

CREATE TABLE Policy(Policy_ID Int,PolicyNumber INT(20) PRIMARY KEY AUTO_INCREMENT, PolicyEffecttiveDate DATETIME, PolicyExpireDate DATETIME, PolicyTimePeriod VARCHAR(100),TotalAmount DOUBLE,TypeOfPolicy VARCHAR(30),FOREIGN KEY(Policy_ID) REFERENCES login(User_Id));

CREATE TABLE Vehicle(ID INT,VehicleNumber INT(20) PRIMARY KEY AUTO_INCREMENT,YEAR INT NOT NULL,Model VARCHAR(50) NOT Null,Color VARCHAR(50) NOT NULL,Mileage INT NOT NULL,VINNumber VARCHAR(20) NOT NULL,VehicleNumberPlate VARCHAR(20) NOT NULL,VehicleRegisteredState VARCHAR(50) NOT NULL,Vehicletype VARCHAR(50),VehicleInsuranceType VARCHAR(50) NOT NULL,FOREIGN KEY(ID) REFERENCES login(User_Id));

CREATE TABLE Agent(Agent_ID INT PRIMARY KEY,Name VARCHAR(50) NOT NULL,Gender CHAR(1) NOT NULL,DOB DATE NOT NULL,Address VARCHAR(70) NOT NULL,Pincode INT(10) NOT NULL,Contact_Number BIGINT(19) NOT NULL,Mother_Name VARCHAR(150) NOT NULL,Father_Name VARCHAR(150) NOT NULL,FOREIGN KEY(Agent_ID) REFERENCES login(User_Id));
